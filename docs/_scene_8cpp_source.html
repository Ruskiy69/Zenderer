111 bool CScene::Render()
112 {
113  // Called every frame because there is no more appropriate
114  // time to call it. Things won't be offloaded multiple times.
115  auto i = m_allEntities.begin(), j = m_allEntities.end();
116 
117  // If the first is offloaded, the rest probably are too.
118  if(i != j && !(*i)->Offloaded())
119  {
120  for( ; i != j; ++i)
121  (*i)->Offload(m_Geometry, false);
122  }
123 
124  m_Geometry.Offload();
125 
126  // Clear our frame buffers from the last drawing.
127  // We will be rendering to FBO1 at first.
128  color4f_t Clear = color4f_t(0.0, 0.0, 0.0, 0.0);
129  m_FBO2.Bind();
130  if(m_through) m_FBO2.Clear(Clear); else m_FBO2.Clear();
131  m_FBO1.Bind();
132  if(m_through) m_FBO1.Clear(Clear); else m_FBO1.Clear();
133 
134  // Set the standard blending state.
135  bool blend = CRenderer::BlendOperation(BlendFunc::IS_ENABLED);
136  CRenderer::BlendOperation(BlendFunc::STANDARD_BLEND);
137 
138  // All geometry is stored here.
139  m_Geometry.Bind();
140 
141  // Prepare for primitive rendering.
142  const CMaterial& M = CRenderer::GetDefaultMaterial();
143  CEffect& E = const_cast<CMaterial&>(M).GetEffect();
144  M.Enable();
145 
146  // Commence individual primitive rendering.
147  i = m_allEntities.begin();
148  for( ; i != j; ++i)
149  {
150  // Adjust for the camera.
151  (*i)->Move((*i)->GetPosition() + m_Camera);
152 
153  // Set the matrix for transformation.
154  const math::matrix4x4_t& Tmp = (*i)->GetTransformation();
155  E.SetParameter("mv", Tmp);
156 
157  auto a = (*i)->cbegin(), b = (*i)->cend();
158  for( ; a != b; ++a)
159  {
160  (*a)->GetMaterial().Enable();
161  (*a)->Draw(true);
162  }
163 
164  // Move back to original position.
165  (*i)->Move((*i)->GetPosition() - m_Camera);
166  }
167 
168  M.Disable();
169 
170  // Shortcut reference.
171  gfxcore::CVertexArray& FS = CRenderer::GetFullscreenVBO();
172 
173  // Primitive rendering is complete.
174  // Now, render lights with additive blending.
175  GLuint final_texture = m_FBO1.GetTexture();
176 
177  if(m_lighting)
178  {
179  // We want to render to the entire scene.
180 
181  // Lighting renders ON the geometry, so we
182  // bind the FBO1 texture to render onto. The
183  // final result ends up on the FBO2 texture.
184  m_FBO2.Bind();
185  CRenderer::EnableTexture(final_texture);
186  CRenderer::BlendOperation(BlendFunc::ADDITIVE_BLEND);
187 
188  auto i = m_allLights.begin(),
189  j = m_allLights.end();
190 
191  for( ; i != j; ++i)
192  {
193  CLight& L = *(*i);
194  L.Enable();
195 
196  // Move with scene
197  L.Adjust(m_Camera.x, m_Camera.y);
198 
199  FS.Draw();
200 
201  // Restore state
202  L.Adjust(-m_Camera.x, -m_Camera.y);
203  L.Disable();
204  }
205 
206  final_texture = m_FBO2.GetTexture();
207  CRenderer::BlendOperation(BlendFunc::STANDARD_BLEND);
208  }
209 
210  // Ping-pong post-processing effects.
211  // This means one is drawn to FBO1, then FBO1's result
212  // is used to draw on FBO2, etc.
213  if(m_ppfx)
214  {
215  // If there was lighting, the first texture is
216  // the second FBO.
217  CRenderTarget& One = m_lighting ? m_FBO2 : m_FBO1;
218  CRenderTarget& Two = m_lighting ? m_FBO1 : m_FBO2;
219 
220  auto i = m_allPPFX.begin(),
221  j = m_allPPFX.end();
222 
223  for(size_t c = 0; i != j; ++i, ++c)
224  {
225  bool even = ((c & 0x1) == 0);
226  if(even) Two.Bind();
227  else One.Bind();
228 
229  (*i)->Enable();
230  CRenderer::EnableTexture(final_texture);
231  FS.Draw();
232  (*i)->Disable();
233 
234  final_texture = even ? Two.GetTexture() : One.GetTexture();
235  }
236  }
237 
238  // Now we have the final scene data in `final_texture`
239 
240  // Doesn't matter which we disable.
241  m_FBO1.Unbind();
242 
243  E.Enable();
244  CRenderer::EnableTexture(final_texture);
245  if(!m_through) CRenderer::BlendOperation(BlendFunc::DISABLE_BLEND);
246  GL(glDisable(GL_DEPTH_TEST));
247 
248  // Make sure the right data is set.
249  E.SetParameter("mv", math::matrix4x4_t::GetIdentityMatrix());
250  E.SetParameter("proj", CRenderer::GetProjectionMatrix());
251 
252  FS.Draw();
253 
254  CRenderer::EnableTexture(0);
255  CRenderer::ResetMaterialState();
256 
257  if(!blend && m_through) CRenderer::BlendOperation(BlendFunc::DISABLE_BLEND);
258 
259  return true;
260 }
