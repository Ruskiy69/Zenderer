131  bool Render(obj::CEntity& Ent, const string_t text = "")
132  {
133  const string_t& text = to_render.empty() ? m_str.str() : to_render;
134  if(text.empty() || !m_loaded) return false;
135 
136  // Fresh start.
137  Ent.Destroy();
138 
139  // 4 vertices for each character, and 6 indices for
140  // each character.
141  uint16_t vlen = text.length() << 2;
142  uint16_t ilen = text.length() * 6;
143 
144  // Create empty buffers.
145  gfxcore::vertex_t* verts = new gfxcore::vertex_t[vlen];
146  gfxcore::index_t* inds = new gfxcore::index_t [ilen];
147 
148  math::vectoru16_t totals;
149 
150  // Rendering position. By default, we use the line height, because the any
151  // line of the given text will at the very most be as tall as a line. But,
152  // it may be true that there are no full line-height characters, so if there
153  // are none then we just start at the lowest one.
154  math::vector_t Pos(
155  0.0,
156  math::min<uint16_t>(m_height, this->GetTextHeight(text))
157  );
158 
159  // Fill up the buffers.
160  for(size_t i = 0; i < vlen; i += 4)
161  {
162  char c = text[i >> 2];
163 
164  // Handle newlines by resetting the x-coordinate and
165  // increasing the y by the current font faces line height
166  // property (universal on all glyphs).
167  if(c == '\n')
168  {
169  Pos.x = 0.0;
170  Pos.y += m_height;
171  continue;
172  }
173 
174  // Use space (' ') as a placeholder for non-renderable chars.
175  char letter = (c > '~' || c < ' ') ? ' ' : c;
176 
177  // Shortcut to glyph data.
178  const glyph_t& gl = m_glyphData[letter];
179 
180  real_t x = gl.position.x; // Store current x-coordinate.
181  real_t h = gl.position.y; // Store current y-coordinate.
182  Pos.x += gl.advance; // Increment position for the next glyph.
183 
184  /*
185  * [i] : top left
186  * [i + 1] : top right
187  * [i + 2] : bottom right
188  * [i + 3] : bottom left
189  */
190  verts[i].position = math::vector_t(Pos.x, m_height - gl.position.x);
191  verts[i+1].position = math::vector_t(Pos.x + gl.size.x, m_height - gl.position.x);
192  verts[i+2].position = math::vector_t(Pos.x + gl.size.x, m_height);
193  verts[i+3].position = math::vector_t(Pos.x, m_height);
194 
195  // Load up the bitmap texture coordinates moving
196  // counter-clockwise from the origin.
197  verts[i].tc = math::vector_t(0, 1);
198  verts[i+1].tc = math::vector_t(1, 1);
199  verts[i+2].tc = math::vector_t(1, 0);
200  verts[i+3].tc = math::vector_t(0, 0);
201 
202  // Uniform font color.
203  for(size_t j = i; j < i + 4; ++j) verts[j].color = m_Color;
204 
205  // Calculate index starting point. Since 6 indices / vertex,
206  // we do i / 4 (since 4 iterations per vertex) to get the
207  // vertex number, and * 6 to get to the first index for it.
208  int x = (i >> 2) * 6;
209 
210  // Standard quad indices.
211  inds[x] = i;
212  inds[x+1] = i + 1;
213  inds[x+2] = i + 3;
214  inds[x+3] = i + 3;
215  inds[x+4] = i + 2;
216  inds[x+5] = i + 1;
217

219  totals.x += math::max<uint16_t>(gl.size.w, gl.advance);
220  totals.y = math::max<uint16_t>(totals.y, gl.size.h);

354  glyph_t LoadGlyphIntoAtlas(const char c, const uint16_t index, const math::vector_t& Pos)
355  {
356  FT_Glyph g;
357 
358  // Render as a monochrome bitmap into glyph struct.
359  FT_Load_Glyph(m_FontFace, index, FT_LOAD_RENDER);
360  FT_Render_Glyph(m_FontFace->glyph, FT_RENDER_MODE_NORMAL);
361  FT_Get_Glyph(m_FontFace->glyph, &g);
362 
363  // Minor shortcuts.
364  FT_GlyphSlot slot = m_FontFace->glyph;
365  FT_Bitmap& bitmap = slot->bitmap;
366 
367  // Shortcut to dimensions.
368  uint32_t w = bitmap.width;
369  uint32_t h = bitmap.rows;
370 
371  if(w == 0 || h == 0)
372  {
373  m_Log << m_Log.SetMode(LogMode::ZEN_ERROR)
374  << "Empty character bitmap for '" << c << '\'' << CLog::endl;
375  }
376 
377  // Create the OpenGL texture and store the FreeType bitmap
378  // in it as a monochrome texture.
379 
380  // Store for restoring later.
381  GLint pack;
382  GL(glGetIntegerv(GL_UNPACK_ALIGNMENT, &pack));
383  GL(glPixelStorei(GL_UNPACK_ALIGNMENT, 1));
384 
385  GL(glTexSubImage2D(GL_TEXTURE_2D, 0, Position.x, Position.y,
386  w, h, GL_RGBA, GL_UNSIGNED_BYTE, bitmap.buffer));
387 
388  GL(glPixelStorei(GL_UNPACK_ALIGNMENT, pack));
389 
390  glyph_t gl;
391  gl.AtlasMap = math::rect_t(Position.x, Position.y, w, h);
392  gl.position = math::vector_t(slot->metrics.horiBearingY >> 6,
393  slot->metrics.horiBearingX >> 6);
394  gl.advance = slot->advance.x >> 6;
395 
396  // Clean up TTF data.
397  FT_Done_Glyph(g);
398 
399  return gl;
400  }
