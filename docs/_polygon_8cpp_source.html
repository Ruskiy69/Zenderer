142 bool zPolygon::Draw(const bool is_bound /*= false*/)
143 {
144  if(mp_VAO == nullptr && !is_bound)
145  {
146  // Create a vertex array and load our data.
147  mp_VAO = new gfxcore::zVertexArray(GL_STATIC_DRAW);
148  mp_VAO->Init();
149  m_offset = mp_VAO->AddData(m_DrawData);
150  if(!mp_VAO->Offload()) return false;
151 
152  // Create our model-view matrix.
153  mp_MVMatrix = new math::matrix4x4_t(math::
154  matrix4x4_t::GetIdentityMatrix());
155 
156  // So we can differentiate between a VAO from a `zScene`
157  // and the one we made ourselves.
158  m_internal = true;
159  }
160 
161  // If something isn't previously bound, we bind the VAO
162  // and the material. If no material, use global default.
163  if(!is_bound)
164  {
165  if(!mp_VAO->Bind()) return false;
166 
167  // Insert our coordinates to transform in the shader.
168  // Ignore the Z coordinate because that's only used for depth
169  // sorting internally anyway and has no effect on visuals.
170  (*mp_MVMatrix)[0][3] = m_Position.x;
171  (*mp_MVMatrix)[1][3] = m_Position.y;
172  //(*mp_MVMatrix)[2][3] = m_Position.z;
173 
174  // Bind our material. If we haven't set one, the default will be used.
175  // We need a default texture because otherwise the color would
176  // always be black due to the way the shader works.
177  const gfx::zEffect& Effect = m_Material.GetEffect();
178  m_Material.Enable();
179 
180  // All effects have these parameters in the vertex shader.
181  if(!Effect.SetParameter("mv", *mp_MVMatrix) ||
182  !Effect.SetParameter("proj", zRenderer::GetProjectionMatrix()))
183  {
184  zRenderer::ResetMaterialState();
185  return false;
186  }
187  }
188 
189  GL(glDrawElements(GL_TRIANGLES, m_DrawData.icount,
190  gfxcore::INDEX_TYPE,
191  (void*)(sizeof(index_t) * m_offset)));
