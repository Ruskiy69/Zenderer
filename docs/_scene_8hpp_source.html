153  bool Render()
154  {
155  // We keep only one matrix instance and just
156  // modify it for every object.
157  math::matrix4x4_t MV(math::matrix4x4_t::GetIdentityMatrix());
158 
159  // Called every frame because there is no more appropriate
160  // time to call it. Things won't be offloaded multiple times.
161  m_Geometry.Offload();
162 
163  // Clear our frame buffers from the last drawing.
164  // We will be rendering to FBO1 at first.
165  m_FBO2.Bind(); m_FBO2.Clear();
166  m_FBO1.Bind(); m_FBO1.Clear();
167 
168  // Set the standard blending state.
169  CRenderer::BlendOperation(BlendFunc::STANDARD_BLEND);
170 
171  // All geometry is stored here.
172  m_Geometry.Bind();
173 
174  // Prepare for primitive rendering.
175  CMaterial& M = CRenderer::GetDefaultMaterial();
176  CEffect& E = M.GetEffect();
177  M.Enable();
178 
179  // Commence individual primitive rendering.
180  auto i = m_allPrimitives.begin(),
181  j = m_allPrimitives.end();
182 
183  for( ; i != j; ++i)
184  {
185  MV.Translate((*i)->GetPosition());
186  MV.TranslateAdj(m_Camera);
187 
188  (*i)->Draw(true);
189  }
190 
191  M.Disable();
192 
193  // Shortcut reference.
194  gfxcore::CVertexArray& FS = CRenderer::GetFullscreenVBO();
195 
196  // Primitive rendering is complete.
197  // Now, render lights with additive blending.
198  GLuint final_texture = m_FBO1.GetTexture();
199 
200  if(m_lighting)
201  {
202  // We want to render to the entire scene.
203 
204  // Lighting renders ON the geometry, so we
205  // bind the FBO1 texture to render onto. The
206  // final result ends up on the FBO2 texture.
207  m_FBO2.Bind();
208  CRenderer::EnableTexture(final_texture);
209  CRenderer::BlendOperation(BlendFunc::ADDITIVE_BLEND);
210 
211  auto i = m_allLights.begin(),
212  j = m_allLights.end();
213 
214  for( ; i != j; ++i)
215  {
216  CLight& L = *(*i);
217  L.Enable();
218 
219  // Move with scene
220  L.Adjust(m_Camera.x, m_Camera.y);
221 
222  FS.Draw();
223 
224  // Restore state
225  L.Adjust(-m_Camera.x, -m_Camera.y);
226  L.Disable();
227  }
228 
229  final_texture = m_FBO2.GetTexture();
230  CRenderer::BlendOperation(BlendFunc::STANDARD_BLEND);
231  }
232 
233  // Ping-pong post-processing effects.
234  // This means one is drawn to FBO1, then FBO1's result
235  // is used to draw on FBO2, etc.
236  if(m_ppfx)
237  {
238  // If there was lighting, the first texture is
239  // the second FBO.
240  CRenderTarget& One = m_lighting ? m_FBO2 : m_FBO1;
241  CRenderTarget& Two = m_lighting ? m_FBO1 : m_FBO2;
242 
243  auto i = m_allPPFX.begin(),
244  j = m_allPPFX.end();
245 
246  for(size_t c = 0; i != j; ++i, ++c)
247  {
248  bool even = ((c & 0x1) == 0);
249  if(even) Two.Bind();
250  else One.Bind();
251 
252  (*i)->Enable();
253  CRenderer::EnableTexture(final_texture);
254  FS.Draw();
255  (*i)->Disable();
256 
257  if(even) final_texture = Two.GetTexture();
258  else final_texture = One.GetTexture();
259  }
260  }
261 
262  // Now we have the final scene data in `final_texture`
263 
264  // Doesn't matter which we disable.
265  m_FBO1.Unbind();
266 
267  E.Enable();
268  CRenderer::EnableTexture(final_texture);
269 
270  // Make sure the right data is set.
271  E.SetParameter("mv", math::matrix4x4_t::GetIdentityMatrix());
272  E.SetParameter("proj", CRenderer::GetProjectionMatrix());
273 
274  FS.Draw();
275 
276  CRenderer::EnableTexture(0);
277  CRenderer::ResetMaterialState();
278 
279  return true;
