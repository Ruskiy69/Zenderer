170  bool Render()
171  {
172  // We keep only one matrix instance and just
173  // modify it for every object.
174  math::matrix4x4_t MV(math::matrix4x4_t::GetIdentityMatrix());
175 
176  // Called every frame because there is no more appropriate
177  // time to call it. Things won't be offloaded multiple times.
178  m_Geometry.Offload();
179 
180  // Clear our frame buffers from the last drawing.
181  // We will be rendering to FBO1 at first.
182  m_FBO2.Bind(); m_FBO2.Clear();
183  m_FBO1.Bind(); m_FBO1.Clear();
184 
185  // Set the standard blending state.
186  CRenderer::BlendOperation(BlendFunc::STANDARD_BLEND);
187 
188  // All geometry is stored here.
189  m_Geometry.Bind();
190 
191  // Prepare for primitive rendering.
192  const CMaterial& M = CRenderer::GetDefaultMaterial();
193  CEffect& E = const_cast<CMaterial&>(M).GetEffect();
194  M.Enable();
195 
196  // Commence individual primitive rendering.
197  auto i = m_allEntities.begin(),
198  j = m_allEntities.end();
199 
200  for( ; i != j; ++i)
201  {
202  MV.Translate((*i)->GetPosition());
203  MV.TranslateAdj(m_Camera);
204 
205  (*i)->Draw(true);
206  }
207 
208  M.Disable();
209 
210  // Shortcut reference.
211  gfxcore::CVertexArray& FS = CRenderer::GetFullscreenVBO();
212 
213  // Primitive rendering is complete.
214  // Now, render lights with additive blending.
215  GLuint final_texture = m_FBO1.GetTexture();
216 
217  if(m_lighting)
218  {
219  // We want to render to the entire scene.
220 
221  // Lighting renders ON the geometry, so we
222  // bind the FBO1 texture to render onto. The
223  // final result ends up on the FBO2 texture.
224  m_FBO2.Bind();
225  CRenderer::EnableTexture(final_texture);
226  CRenderer::BlendOperation(BlendFunc::ADDITIVE_BLEND);
227 
228  auto i = m_allLights.begin(),
229  j = m_allLights.end();
230 
231  for( ; i != j; ++i)
232  {
233  CLight& L = *(*i);
234  L.Enable();
235 
236  // Move with scene
237  L.Adjust(m_Camera.x, m_Camera.y);
238 
239  FS.Draw();
240 
241  // Restore state
242  L.Adjust(-m_Camera.x, -m_Camera.y);
243  L.Disable();
244  }
245 
246  final_texture = m_FBO2.GetTexture();
247  CRenderer::BlendOperation(BlendFunc::STANDARD_BLEND);
248  }
249 
250  // Ping-pong post-processing effects.
251  // This means one is drawn to FBO1, then FBO1's result
252  // is used to draw on FBO2, etc.
253  if(m_ppfx)
254  {
255  // If there was lighting, the first texture is
256  // the second FBO.
257  CRenderTarget& One = m_lighting ? m_FBO2 : m_FBO1;
258  CRenderTarget& Two = m_lighting ? m_FBO1 : m_FBO2;
259 
260  auto i = m_allPPFX.begin(),
261  j = m_allPPFX.end();
262 
263  for(size_t c = 0; i != j; ++i, ++c)
264  {
265  bool even = ((c & 0x1) == 0);
266  if(even) Two.Bind();
267  else One.Bind();
268 
269  (*i)->Enable();
270  CRenderer::EnableTexture(final_texture);
271  FS.Draw();
272  (*i)->Disable();
273 
274  if(even) final_texture = Two.GetTexture();
275  else final_texture = One.GetTexture();
276  }
277  }
278 
279  // Now we have the final scene data in `final_texture`
280 
281  // Doesn't matter which we disable.
282  m_FBO1.Unbind();
283 
284  E.Enable();
285  CRenderer::EnableTexture(final_texture);
286 
287  // Make sure the right data is set.
288  E.SetParameter("mv", math::matrix4x4_t::GetIdentityMatrix());
289  E.SetParameter("proj", CRenderer::GetProjectionMatrix());
290 
291  FS.Draw();
292 
293  CRenderer::EnableTexture(0);
294  CRenderer::ResetMaterialState();
295 
296  return true;
297  }
