104

106 
107  m_Log << m_Log.SetMode(LogMode::ZEN_DEBUG) << m_Log.SetSystem("FreeType")
108  << "Rendering text string '" << text << "'." << CLog::endl;
109 
110  // Fresh start.
111  Ent.Destroy();
112 
113  // 4 vertices for each character, and 6 indices for
114  // each character.
115  uint16_t vlen = text.length() << 2;
116  uint16_t ilen = text.length() * 6;
117 
118  // Create empty buffers.
119  gfxcore::vertex_t* verts = new gfxcore::vertex_t[vlen];
120  gfxcore::index_t* inds = new gfxcore::index_t [ilen];
121 
122  math::vectoru16_t totals;
123  uint16_t tmp_tx = 0;
124 
125  // Rendering position. We calculate this by detecting any newline ('\n')
126  // characters. If there aren't any, the position begins at the (0, 0, 1),
127  // otherwise, it begins at the end of the last line, since rendering is
128  // performed upside down (to be compatible with the quad texture coords).
129  math::vectoru16_t Pos;
130 
131  uint16_t line = 0;
132  auto a = text.begin(), b = text.end();
133  for( ; a != b; ++a)
134  {
135  if(*a == '\n') ++line;
136  const auto it = m_glyphData.find(*a);
137  if(it == m_glyphData.end()) continue;
138  Pos.y = math::max<uint16_t>(Pos.y,
139  it->second.size.y - it->second.position.y);
140  }
141 
142  // Fill up the buffers.
143  for(size_t i = 0; i < vlen; i += 4)
144  {
145  char c = text[i >> 2];
146 
147  // Handle newlines by resetting the x-coordinate and
148  // increasing the y by the current font faces line height
149  // property (universal on all glyphs).
150  if(c == '\n')
151  {
152  Pos.x = 0; Pos.y += m_height;
153  totals.x = math::max(tmp_tx, totals.x);
154  tmp_tx = 0;
155  --line;
156  continue;
157  }
158 
159  // Use space (' ') as a placeholder for non-renderable chars.
160  char letter = (c > '~' || c < ' ') ? ' ' : c;
161 
162  // Shortcut to glyph data.
163  const glyph_t& gl = m_glyphData[letter];
164 
165  if(letter == ' ')
166  {
167  // Increment position for the next glyph.
168  tmp_tx += gl.advance;
169  Pos.x += gl.advance;
170  continue;
171  }
172 
173  /*
174  * [i] : top left
175  * [i + 1] : top right
176  * [i + 2] : bottom right
177  * [i + 3] : bottom left
178  */
179  uint16_t start_y = Pos.y + gl.position.y;
180  verts[i].position = math::vector_t(Pos.x, start_y - gl.size.y);
181  verts[i+1].position = math::vector_t(Pos.x + gl.size.x, start_y - gl.size.y);
182  verts[i+2].position = math::vector_t(Pos.x + gl.size.x, start_y);
183  verts[i+3].position = math::vector_t(Pos.x, start_y);
184 
185  // Load up the bitmap texture coordinates moving
186  // counter-clockwise from the top-left.
187  verts[i].tc = math::vector_t(0, 1);
188  verts[i+1].tc = math::vector_t(1, 1);
189  verts[i+2].tc = math::vector_t(1, 0);
190  verts[i+3].tc = math::vector_t(0, 0);
191 
192  // Uniform font color.
193  for(size_t j = i; j < i + 4; ++j) verts[j].color = m_Color;
194 
195  // Calculate index starting point. Since 6 indices per vertex,
196  // we do i / 4 (since 4 iterations per vertex) to get the
197  // vertex number, and * 6 to get to the first index for it.
198  int x = (i >> 2) * 6;
199 
200  // Standard quad indices.
201  inds[x] = i;
202  inds[x+1] = i + 1;
203  inds[x+2] = i + 3;
204  inds[x+3] = i + 3;
205  inds[x+4] = i + 2;
206  inds[x+5] = i + 1;
207 
208  // Track total dimensions.
209  tmp_tx += math::max(gl.size.x, gl.advance);
210  totals.y = math::max<uint16_t>(totals.y, m_height * line + gl.size.y);
211 
212  // Increment position for the next glyph.
213  Pos.x += gl.advance;
214  }
215

217

219  // then assign that texture to the entity.

221  gfxcore::CVertexArray VAO(GL_STATIC_DRAW);
222  gfx::CRenderTarget FBO(totals.x, totals.y);
223

225  D.vcount = vlen;
226  D.Indices = inds;
227  D.icount = ilen;
228

230  {
231  delete[] verts;
232  delete[] inds;
233  return false;
234  }
235 
236  VAO.AddData(D);
237  VAO.Offload();
238 
239  VAO.Bind();
240  FBO.Bind();
241

243  gfxcore::BlendFunc::IS_ENABLED);
244 
245  gfxcore::CRenderer::BlendOperation(gfxcore::BlendFunc::STANDARD_BLEND);
246  s_FontFx->Enable();
247  s_FontFx->SetParameter("proj", gfxcore::CRenderer::GetProjectionMatrix());
248

251  {
252  // Render each character (skip if unrenderable).
253  if(text[i] > '~' || text[i] <= ' ') continue;
254

270

272  gfxcore::CTexture* pTexture =

278

284  // of scope soon, thus destroying the texture handle, but the data needs

293  delete[] inds;
294

298

305

332 {
333  ZEN_ASSERTM(mp_Assets != nullptr, "an asset manager must be attached");
334  FT_Glyph g;
335 
336  // Render as a monochrome bitmap into glyph struct.
337  FT_Load_Glyph(m_FontFace, index, FT_LOAD_RENDER);
338  FT_Render_Glyph(m_FontFace->glyph, FT_RENDER_MODE_NORMAL);
339  FT_Get_Glyph(m_FontFace->glyph, &g);
340 
341  // Minor shortcuts.
342  FT_GlyphSlot slot = m_FontFace->glyph;
343  FT_Bitmap& bitmap = slot->bitmap;
344 
345  // Shortcut to dimensions.
346  uint32_t w = bitmap.width;
347  uint32_t h = bitmap.rows;
348 
349  if(w == 0 || h == 0)
350  {
351  m_Log << m_Log.SetMode(LogMode::ZEN_ERROR)
352  << "Empty character bitmap for '" << c << '\'' << CLog::endl;
353  }
354 
355  // Create the OpenGL texture and store the FreeType bitmap
356  // in it as a monochrome texture.
357 
358  // Store for restoring later.
359  GLint pack;
360  GL(glGetIntegerv(GL_UNPACK_ALIGNMENT, &pack));
361  GL(glPixelStorei(GL_UNPACK_ALIGNMENT, 1));
362 
363  gfxcore::CTexture* pTexture = mp_Assets->Create<gfxcore::CTexture>(this->GetOwner());
364  pTexture->LoadFromRaw(GL_R8, GL_RED, w, h, bitmap.buffer);
365  std::stringstream ss;
366  ss << "Size " << m_size << " texture bitmap for '" << c << '\'';
367  pTexture->SetFilename(ss.str());
368 
369  GL(glPixelStorei(GL_UNPACK_ALIGNMENT, pack));
370 
371  // Store the glyph internally.
372  glyph_t glyph;
373  glyph.texture = pTexture;
374  glyph.size = math::Vector<uint32_t>(w, h);
375  glyph.position = math::Vector<int32_t>(slot->metrics.horiBearingX >> 6,
376  slot->metrics.horiBearingY >> 6);
377  glyph.advance = slot->advance.x >> 6;
378  m_glyphData[c] = glyph;
379 
380  // Clean up TTF data.
381  FT_Done_Glyph(g);
382 
383  return true;
384 }
