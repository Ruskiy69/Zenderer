714 bool parse_msg(const string_t& data, PongPacket& P)
715 {
716  P.seq = P.stamp = P.size = 0;
717  P.type = PacketType::UNKNOWN;
718  P.data = "";
719 
720  if(data.empty() || data.size() < MIN_PONG) return false;
721 
722  std::vector<string_t> parts = util::split(data, ':');
723  if(parts.size() < 5) return false;
724 
725  P.seq = stoi(parts[0]);
726  P.stamp= stoi(parts[1]);
727  P.type = static_cast<PacketType>(stoi(parts[2]));
728  P.size = stoi(parts[3]);
729  P.data.resize(data.size(), '\0');
730 
731  // Compensate for the fact that the data itself may have
732  // contained the ':' character.
733  string_t final_data;
734  std::for_each(parts.begin() + 4, parts.end(),
735  [&final_data](const string_t& s) {
736  final_data.insert(final_data.end(), s.begin(), s.end());
737  });
738  P.data = final_data;
739 
740  return P.type != PacketType::UNKNOWN;
741 }
