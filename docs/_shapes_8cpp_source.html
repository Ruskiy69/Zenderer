157 {
158  real_t denom = (T[1].y - T[2].y) * (T[0].x - T[2].x)
159  + (T[2].x - T[1].x) * (T[0].y - T[2].y);
160

162  if(compf(denom, 0.0)) return true;
163  denom = 1.0 / denom;
164

166  (T[2].x - T[1].x) * (V.y - T[2].y));
167  if(alpha < 0) return false;
