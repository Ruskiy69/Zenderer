143 void CSceneManager::SortByMaterial(
144  const std::vector<ent::CEntity*>& allObjects,
145  std::map<uint32_t, std::list<ent::CEntity*>>& groupedObjects)
146 {
147  auto start = groupedObjects.begin(), end = groupedObjects.end();
148 
149  // Iterate over each material and check that the entities for
150  // this material haven't changed. If they have, insert them into
151  // the list for their proper material.
152  for( ; start != end; ++start)
153  {
154  auto i = start->second.begin(), j = start->second.end();
155  for( ; i != j; )
156  {
157  auto id = CSorter::ExtractMaterialID(*i);
158 
159  // We are already in the right place.
160  if(start->first == id)
161  {
162  ++i;
163  continue;
164  }
165 
166  auto finder = groupedObjects.find(id);
167  if(finder == groupedObjects.end())
168  {
169  groupedObjects[id] = std::list<ent::CEntity*>>(1, *i);
170  }
171  else
172  {
173  finder->second.push_back(*i);
174  }
175 
176  i = start->second.erase(i);
177  }
178  }
